  <script>
    document.getElementById("requestForm").addEventListener("submit", async function(e){
      e.preventDefault();

      const formData = {
        type: document.getElementById("type").value,
        title: document.getElementById("title").value,
        category: document.getElementById("category").value,
        country: document.getElementById("country").value,
        offers: parseInt(document.getElementById("offers").value),
        date: new Date().toISOString().split("T")[0]
      };

      try {
        // Mevcut verileri al
        const res = await fetch("../data/requests.json");
        const data = await res.json();

        // Yeni veriyi ekle
        data.push(formData);

        // JSON olarak kaydet (GitHub Pages'de doğrudan yazma olmayabilir ama localde test edilebilir)
        await fetch("../data/requests.json", {
          method: "PUT",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(data, null, 2)
        });

        document.getElementById("message").innerText = "Request successfully added!";
        e.target.reset();

      } catch (err) {
        console.error(err);
        document.getElementById("message").innerText = "Unable to save (read-only mode).";
      }
    });
  </script>
